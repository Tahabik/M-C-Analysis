  \begin{frame}{Background cont.}
      \begin{itemize}
          \item \textbf{Hardware-Assisted Virtualization:}
          \begin{itemize}
              \item Powered by KVM in the Linux kernel for virtualization on the same host architecture.
              \item Processing is done on actual hardware via CPU Virtualization technologies (VT-x/AMD-V).
          \end{itemize}
          \item \textbf{Emulation:} 
          \begin{itemize}
            \item For running OSes that run on entirely different CPU architectures!
            \item Powered by QEMU's TCG (Tiny Code Generator). 
          \end{itemize}
      \end{itemize}
  \end{frame}
  \begin{frame}{Motivation: The Virtual Memory Challenge}
      \begin{columns}[T]
          % Left Column: Concepts & Overheads
          \begin{column}{0.55\textwidth}
              \textbf{The Guest Memory Layering Illusion:}
              \begin{itemize}
                  \item Guest OS maps Guest Virtual Addresses (GVA) to Guest Physical Addresses (GPA).
                  \item Host OS must map GPA to final Host Physical Addresses (HPA).
              \end{itemize}
              % \vspace{0.3cm}
          \end{column}
          
          \begin{column}{0.45\textwidth}
              \textbf{Two Architectural Solutions:}
              \begin{itemize}
                  \item \textbf{Hardware EPT (KVM):} The hardware MMU natively walks both page tables simultaneously in silicon. High-speed, but TLB misses are costly.
                  \item \textbf{Software MMU (TCG):} QEMU maintains an inline software TLB cache. If it misses, it triggers a slow, multi-layered C helper routine loop.
              \end{itemize}
          \end{column}
      \end{columns}
  \end{frame}

  \begin{frame}{Motivation: Instruction Pipeline Divergence}
      \textbf{Instruction Handling Disparity:}
      \begin{columns}[T]
          \begin{column}{0.45\textwidth}
              \begin{itemize}
                  \item \textbf{KVM Direct Execution:} Guest instructions execute natively on physical execution units at full hardware frequency. 
                  \item Overheads only trigger during trapping instructions (\textit{VM-Exits}) like I/O or hypercalls.
              \end{itemize}
          \end{column}

          % Right Column: Execution Overheads
          \begin{column}{0.55\textwidth}
              \vspace{0.2cm}
              \textbf{TCG Binary Translation:}
              \begin{itemize}
                  \item Guest binary blocks are completely intercepted, compiled to Intermediate Representation (IR), and rewritten into Host Assembly.
                  \item Code changes require synchronous, costly generation cache invalidations (\textit{SMC Penalties}).
              \end{itemize}
          \end{column}
      \end{columns}
  \end{frame}
