  \begin{frame}{Background cont.}
      \begin{itemize}
          \item \textbf{Hardware-Assisted Virtualization:}
          \begin{itemize}
              \item Powered by KVM in the Linux kernel for virtualization on the same host architecture.
              \item Processing is done on actual hardware via CPU Virtualization technologies (VT-x/AMD-V).
          \end{itemize}
          \item \textbf{Emulation:} 
          \begin{itemize}
            \item For running OSes that run on entirely different CPU architectures!
            \item Powered by QEMU's TCG (Tiny Code Generator). 
          \end{itemize}
      \end{itemize}
  \end{frame}
  \begin{frame}{Motivation: The Virtual Memory Challenge}
      \begin{columns}[T]
          % Left Column: Concepts & Overheads
          \begin{column}{0.55\textwidth}
              \textbf{The Guest Memory Layering Illusion:}
              \begin{itemize}
                  \item Guest OS maps Guest Virtual Addresses (GVA) to Guest Physical Addresses (GPA).
                  \item Host OS must map GPA to final Host Physical Addresses (HPA).
              \end{itemize}
              % \vspace{0.3cm}
          \end{column}
          
          \begin{column}{0.45\textwidth}
              \textbf{Two Architectural Solutions:}
              \begin{itemize}
                  \item \textbf{Hardware EPT (KVM):} The hardware MMU natively walks both page tables simultaneously in silicon. High-speed, but TLB misses are costly.
                  \item \textbf{Software MMU (TCG):} QEMU maintains an inline software TLB cache. If it misses, it triggers a slow, multi-layered C helper routine loop.
              \end{itemize}
          \end{column}
      \end{columns}
  \end{frame}

  \begin{frame}{Motivation: Instruction Pipeline Divergence}
      \textbf{Instruction Handling Disparity:}
      \begin{columns}[T]
          \begin{column}{0.45\textwidth}
              \begin{itemize}
                  \item \textbf{KVM Direct Execution:} Guest instructions execute natively on physical execution units at full hardware frequency. 
                  \item Overheads only trigger during trapping instructions (\textit{VM-Exits}) like I/O or hypercalls.
              \end{itemize}
          \end{column}

          % Right Column: Execution Overheads
          \begin{column}{0.55\textwidth}
              \vspace{0.2cm}
              \textbf{TCG Binary Translation:}
              \begin{itemize}
                  \item Guest binary blocks are completely intercepted, compiled to Intermediate Representation (IR), and rewritten into Host Assembly.
                  \item Code changes require synchronous, costly generation cache invalidations (\textit{SMC Penalties}).
              \end{itemize}
          \end{column}
      \end{columns}
  \end{frame}

\begin{frame}{Project Goal: Quantifying the Virtualization Tax}
      \begin{itemize}
          \item \textbf{The Main Objective:} Isolate and measure the exact architectural overheads of hardware-assisted memory management (KVM/EPT) versus software emulation (QEMU/SoftMMU).
          
          \vspace{0.4cm}
          \item \textbf{The Telemetry Stack:} Utilize Linux profiling tools to expose cycle-level bottlenecks:
          \begin{itemize}
              \item \texttt{perf stat} for hardware counters (cache misses, dTLB stalls).
              \item \texttt{perf record} and \texttt{perf meme} to map execution hotspots.
              \item \textbf{Flamegraphs} to visually pinpoint wasted CPU cycles in the virtualization stack.
          \end{itemize}
          
          \vspace{0.4cm}
          \item \textbf{Workload Strategy:} Deploy targeted benchmarks (Pointer Chaser, Matrix Multiplication, Self-Modifying Code) to explicitly stress memory latency, compute boundaries, and JIT translation penalties.
      \end{itemize}
\end{frame}

 \begin{frame}{Workload Methodology: The Benchmarks}
      \begin{itemize}
          \item \textbf{Pointer Chaser (Memory Latency Bound):}
          \begin{itemize}
              \item Traverses a linked list with randomized and sequential memory allocations.
              \item \textbf{Stress Profile:} Maximizes dTLB misses and forces constant cache eviction to evaluate the raw penalty of EPT page walks versus SoftMMU table parsing.
          \end{itemize}
          
          \vspace{0.3cm}
          \item \textbf{Matrix Multiplication (Compute \& Cache Bound):}
          \begin{itemize}
              \item Performs dense linear algebra operations ($O(n^3)$ complexity).
              \item \textbf{Stress Profile:} Exhibits high spatial and temporal data locality. Tests how well each subsystem handles streaming data across L1, L2, and L3 cache boundaries.
          \end{itemize}

          \vspace{0.3cm}
          \item \textbf{Self-Modifying Code (Translation Bound):}
          \begin{itemize}
              \item Generates or alters instructions at runtime before executing them.
              \item \textbf{Stress Profile:} Directly forces the invalidation of JIT-compiled translation blocks. Exposes the execution penalty of QEMU TCG code generation cache flushes.
          \end{itemize}
      \end{itemize}
  \end{frame}

      \vspace{0.3cm}
      \textbf{Key Architectural Takeaways:}
      \begin{itemize}
          \item \textbf{Instruction Expansion:} TCG generates massive instruction bloat to maintain software MMU hash tables compared to KVM's native 2D page walk.
          \item \textbf{Pipeline Pollution:} TCG dynamic translation completely destroys instruction locality, resulting in 100x more instruction cache misses.
          \item \textbf{The Memory Wall:} Both virtualization strategies hit severe bottlenecks at 16MB, indicating host L3 cache overflow.
      \end{itemize}
    \end{frame}

    \begin{frame}{Phase 2 Setup: Contextual Latency Profiling}
      \begin{columns}[T]
          % Left Column: The Tool
          \begin{column}{0.5\textwidth}
              \textbf{The Tool: \texttt{perf mem}}
              \vspace{0.2cm}
              \begin{itemize}
                  \item \textbf{Beyond Counting:} Shifts from counting raw misses to capturing the exact \textit{latency} (cycles) and \textit{source} (L1, L2, RAM) of specific memory loads.
                  \item \textbf{High-Latency Filtration:} We configure the tracer to ignore fast L1 hits and strictly log memory accesses taking $>30$ CPU cycles.
              \end{itemize}
